Countdown.js

A simple and easy-to-use countdown script

Getting Started

Learn how to easily start using Countdown.js


Download

Download Stable Version (v0.0.3) [Recomended!].

This version is stable, the configuration and other content will not be changed. It will always be compatible with the code.


Download Unstable Version

This version is a developer version. It may contain bugs and is constantly being improved. Using it in production is not recommended.


Include scripts

It is possible to place one of two versions in the script - minified and unminified JS files.

<script type="text/javascript" src="js/countdown.min.js"></script>

Or:

<script type="text/javascript" src="js/countdown.js"></script>

Set end date

Choose a date that supports JavaScript in whatever format you like.

var endDate = "2050/01/01";

Callback parameters

The callback function has two parameters.

The first parameter contains the following calculations related to the countdown.

Callback parameters
Parametrs Description
seconds Seconds to end date.
minutes Minutes to end date.
hours Hours to end date.
days Days to end date.
daysToWeek Days to selected week.
daysToMonth Days to selected month.
weeks Weeks to end date.
weeksToMonth Weeks to selected month.
months Months to selected date.
monthsToYear Months to selected year.
months Months to selected date.
years Years to selected date.
todalDays Total days to end date.
todalHours Total hours to end date.
todalMinutes Total minutes to end date.
todalSeconds Total seconds to end date.

Example 1. - Advanced

var counterElement = document.querySelector(".counter"); // Select #counter

var secondsTotalElement = document.querySelector("#seconds-total"); // Select #seconds-total
var minutesTotalElement = document.querySelector("#minutes-total"); // Select #minutes-total
var hoursTotalElement = document.querySelector("#hours-total"); // Select #hours-total
var daysTotalElement = document.querySelector("#days-total"); // Select #days-total
var yearsElement = document.querySelector("#years"); // Select #years
var monthsToYearElement = document.querySelector("#months-to-year"); // Select #months-to-year
var monthsElement = document.querySelector("#months"); // Select #months
var weeksToMonthElement = document.querySelector("#weeks-to-month"); // Select #weeks-to-month
var weeksElement = document.querySelector("#weeks"); // Select #weeks
var daysToMonthElement = document.querySelector("#days-to-month"); // Select #days-to-month
var daysToWeekElement = document.querySelector("#days-to-week"); // Select #days-to-week
var daysElement = document.querySelector("#days"); // Select #days
var hoursElement = document.querySelector("#hours"); // Select #hours
var minutesElement = document.querySelector("#minutes"); // Select #minutes
var secondsElement = document.querySelector("#seconds"); // Select #seconds
var endDate = (new Date().getFullYear() + 10) + "/01/01"; // Select date (Default javascript date format)

var myCountDown = new countdown(endDate, function (remaining, finished) { // Setup countdown

if (finished) {
counterElement.textContent = "Expired"; // Show text when expired
} else {
secondsTotalElement.textContent = remaining.totalSeconds;
minutesTotalElement.textContent = remaining.totalMinutes;
hoursTotalElement.textContent = remaining.totalHours;
daysTotalElement.textContent = remaining.totalDays;
yearsElement.textContent = remaining.years;
monthsToYearElement.textContent = remaining.monthsToYear;
monthsElement.textContent = remaining.months;
weeksToMonthElement.textContent = remaining.weeksToMonth;
weeksElement.textContent = remaining.weeks;
daysToMonthElement.textContent = remaining.daysToMonth;
daysToWeekElement.textContent = remaining.daysToWeek;
daysElement.textContent = remaining.days;
hoursElement.textContent = remaining.hours;
minutesElement.textContent = remaining.minutes;
secondsElement.textContent = remaining.seconds;
}
});

var endDate = (new Date().getFullYear()) + "/08/02"; // Select date (Default javascript date format)

var myCountDown = new countdown(endDate, function (remaining, finished) { // Setup countdown

if (finished) {
containerElement.textContent = "Expired"; // Show text when expired
}

daysElement.textContent = remaining.totalDays; // Render message
hoursElement.textContent = remaining.hours;

});
